\subsection{Divergence Asymmetry and Time}
\label{subsec:divergenceAsymmetryAndTime}

\begin{definition}[Temporal Asymmetry Functional]
\label{def:temporalFunctional}
Define the temporal asymmetry functional:
\begin{equation}
\mathcal{A}[\psi, \phi] := D[\psi \| \phi] - D[\phi \| \psi],
\end{equation}
where $D[\psi \| \phi]$ is the Bregman divergence from Definition \ref{def:bregman}. This quantifies the asymmetry of the divergence structure, breaking time-reversal symmetry and encoding the arrow of time.
\end{definition}

\begin{lemma}[Properties of Asymmetry Functional]
\label{lem:asymmetryProperties}
The functional $\mathcal{A}$ satisfies:

\begin{enumerate}
\item \textbf{Antisymmetry:} $\mathcal{A}[\psi, \phi] = -\mathcal{A}[\phi, \psi]$.

\item \textbf{Vanishing for equal arguments:} $\mathcal{A}[\psi, \psi] = 0$.

\item \textbf{Third-order expansion:} For small perturbations $h, k \in \mathcal{H}$ around $\phi_0$:
\begin{equation}
\mathcal{A}[\phi_0 + h, \phi_0 + k] = \int_X V'''(|\phi_0|^2) \text{Im}(\overline{\phi_0} \cdot h) \text{Im}(\overline{\phi_0} \cdot k) |\phi_0|^2 d\mu + O(\|h\|^4, \|k\|^4).
\end{equation}

\item \textbf{Non-degeneracy on generic configurations:} For $\psi \neq \phi$ with generic configurations, $\mathcal{A}[\psi, \phi] \neq 0$ unless $\psi$ and $\phi$ lie on a null divergence surface (measure-zero set).
\end{enumerate}
